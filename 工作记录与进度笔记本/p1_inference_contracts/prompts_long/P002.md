```python
class PendingQueue:
    def __init__(self):
        self.items = []

    def add(self, request):
        request.enqueue_ns = monotonic_ns()
        self.items.append(request)

    def form_batch(self, max_tokens):
        batch = []
        total = 0
        while self.items and total + self.items[0].prompt_tokens <= max_tokens:
            req = self.items.pop(0)
            req.dequeue_ns = monotonic_ns()
            batch.append(req)
            total += req.prompt_tokens
        return batch
```
