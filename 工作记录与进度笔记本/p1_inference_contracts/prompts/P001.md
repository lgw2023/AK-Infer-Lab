```python
def summarize_latency(samples_us):
    samples = sorted(samples_us)
    if not samples:
        return {"count": 0, "p50": None, "p99": None}
    p50_index = int((len(samples) - 1) * 0.50)
    p99_index = int((len(samples) - 1) * 0.99)
    return {"count": len(samples), "p50": samples[p50_index], "p99": samples[p99_index]}
```
